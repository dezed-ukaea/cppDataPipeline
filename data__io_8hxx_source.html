68 const toml::value data_{
69 {param_name_, {{"type", "point-estimate"}, {"value", value}}}};
70
71 const ghc::filesystem::path output_filename_ =
72 data_store_ / namespace_ / data_product.parent_path() /
73 std::string(version_num.to_string() + ".toml");
74 std::ofstream toml_out_;
75
76 if (!ghc::filesystem::exists(output_filename_.parent_path())) {
77 ghc::filesystem::create_directories(output_filename_.parent_path());
78 }
